// lora_link_core.h -- shared LoRa "link layer" for ESP32 + SX1262 (RadioLib)
// Header-only, Arduino-friendly. No heap allocations.
// - Unifies DIO1 ISR flagging + state machine (Idle/Tx/Rx)
// - Shared TX scheduler (with optional random backoff)
// - Shared RX window (finite like Master) OR continuous (like Node/WLED)
// - Small MAC / address helpers
//
// How to use (short):
//   1) Board code creates SX1262 radio with its own pins (Module(cs,dio1,rst,busy,spi)).
//   2) Call LoraLink::beginCommon(radio, cfg) with your PHY config (freq, bw, sf, cr, ...).
//      You can leave device-specific options unset: txPowerDbm/diO2RfSwitch/rxBoost have defaults.
//   3) Keep one LoraLink::Core 'll' per device. Attach ISR once via LoraLink::attachDio1(radio, ll).
//   4) Define callbacks (onRxPacket, onTxDone, ...).
//   5) In loop(): LoraLink::service(ll, cb).
//   6) For Master-style RX window: LoraLink::requestRxWindow(ll, windowMs).
//      For Node continuous RX: LoraLink::requestRxWindow(ll, 0) once (or at setup()).
//   7) For sending: schedule with LoraLink::scheduleSend(...) or scheduleSendWithBackoff(...).
//
// Notes:
// - This file intentionally does NOT know your pinout. Keep Module() creation in each firmware.
// - Works with RadioLib SX1262. Requires <RadioLib.h> and <Arduino.h>.
// - Uses a single static ISR trampoline; ok because Master and Node are separate binaries.
// - Keep packets <= 32 bytes here to be safe; adapt if your app needs more.
//
// License: MIT

// -------------------- PHY config with device-specific overrides --------------------
struct PhyCfg {
  float   freqMHz      = 868.0f;
  float   bwKHz        = 125.0f;
  uint8_t sf           = 7;     // LoRa spreading factor
  uint8_t crDen        = 5;     // LoRa coding rate denominator (5 => 4/5)
  uint8_t syncWord     = 0x12;
  uint16_t preamble    = 8;
  bool    crcOn        = true;

  // Device-specific knobs (optional). If left "unset", defaults apply.
  // For txPowerDbm: INT8_MIN means "use default 14 dBm".
  int8_t  txPowerDbm   = INT8_MIN;

  // Tri-state flags: -1 = don't touch / leave RadioLib default, 0 = disable, 1 = enable
  int8_t  dio2RfSwitch = -1;    // radio.setDio2AsRfSwitch(...)
  int8_t  rxBoost      = -1;    // radio.setRxBoostedGainMode(...)
};

// Forward declaration
struct Core;

// -------------------- Callbacks --------------------
struct Callbacks {
  void (*onRxPacket)(const uint8_t* pkt, uint8_t len, int16_t rssi, int8_t snr, void* ctx) = nullptr;
  void (*onTxStart)(void* ctx) = nullptr;
  void (*onTxDone)(void* ctx) = nullptr;
  void (*onRxWindowOpen)(uint16_t ms, void* ctx) = nullptr;
  void (*onRxWindowClosed)(uint16_t rxCountDelta, void* ctx) = nullptr;
  void* ctx = nullptr;
};


// -------------------- Mode --------------------
enum class Mode : uint8_t { Idle, Tx, Rx };
enum class RxKind : uint8_t { None, Timed, Continuous };
enum class TxArbiter : uint8_t { None, CadNeeded, CadPending };

// -------------------- Core state --------------------
struct Core {
  SX1262*   radio         = nullptr;
  volatile bool dio1Flag  = false;

  Mode      current       = Mode::Idle;     // Idle, Tx, Rx
  // --- RX-Status ---
  RxKind    rxKind        = RxKind::None;   // aktueller RX-Typ (nur wenn current==Rx)
  uint32_t  rxWindowEndMs = 0;              // nur für RxTimed
  uint16_t  rxCount       = 0;
  uint16_t  rxCountWinStart = 0;

  // --- RX-Request (Wunsch) ---
  RxKind    reqRxKind     = RxKind::None;   // gewünschter RX-Typ
  uint16_t  reqRxMs       = 0;              // nur wenn reqRxKind==Timed

  // --- Default-Modus pro Gerät ---
  RxKind    defaultRxKind = RxKind::None;   // Master: None (Idle), Slave: Continuous
  uint16_t  defaultRxMs   = 0;              // für default Continuous=0 (echt kontinuierlich), für default Timed optional

  // --- TX-Queue ---
  bool      hasPending    = false;
  uint8_t   txBuf[64];
  uint8_t   txLen         = 0;
  uint32_t  earliestTxAtMs= 0;

  // Telemetrie
  int16_t   lastRssi      = 0;
  int8_t    lastSnr       = 0;
  uint32_t  txCount       = 0;

  // --- LBT / Arbiter / ToA ---
  bool       lbtEnable   = false;               // im Setup setzen
  TxArbiter  txArb       = TxArbiter::None;     // LBT-Status
  uint32_t   toaUsMax17  = 0;                   // ToA-Cache (µs) für 17-Byte-Paket

};

// -------------------- Static ISR trampoline --------------------
#if defined(ESP32)
  #define LL_ISR_ATTR IRAM_ATTR
#else
  #define LL_ISR_ATTR
#endif

// Each binary has at most one radio/Core pair ⇒ one trampoline is sufficient.
static Core* volatile g_ll = nullptr;

static void LL_ISR_ATTR onDio1ISR_trampoline() {
  if (g_ll) g_ll->dio1Flag = true;
}

// -------------------- Helpers --------------------
inline bool beginCommon(SX1262& radio, Core& ll, const PhyCfg& cfg) {
  const int8_t power = (cfg.txPowerDbm == INT8_MIN) ? 14 : cfg.txPowerDbm;

  int16_t st = radio.begin(cfg.freqMHz, cfg.bwKHz, cfg.sf, cfg.crDen,
                           cfg.syncWord, power, cfg.preamble);
  if (st != RADIOLIB_ERR_NONE) return false;

  if (cfg.crcOn) radio.setCRC(true); else radio.setCRC(false);
  if (cfg.dio2RfSwitch != -1) radio.setDio2AsRfSwitch(cfg.dio2RfSwitch == 1);
  if (cfg.rxBoost != -1)      radio.setRxBoostedGainMode(cfg.rxBoost == 1);

  radio.standby();        // ensure standby after init

  // NEU: Core <-> Radio verknüpfen und ToA(17) cachen
  ll.radio = &radio;
  ll.toaUsMax17 = radio.getTimeOnAir(17);   // µs

  return true;
}

inline void attachDio1(SX1262& radio, Core& ll) {
  g_ll = &ll;
  radio.setDio1Action(onDio1ISR_trampoline);
}

inline void cacheToA17(Core& ll) {
  if (ll.radio) ll.toaUsMax17 = ll.radio->getTimeOnAir(17); // µs
}

// Maximaler LBT-Backoff in Millisekunden: floor(ToA_17B/1000), mindestens 5 ms
inline uint16_t lbtBackoffMaxMs(const Core& ll) {
  uint32_t ms = ll.toaUsMax17 / 1000U;   // floor(ToA/1000)
  return (ms < 5U) ? 5U : (uint16_t)ms;  // mind. 5 ms
}

// RadioLib RNG: random(min, maxExclusive). Für inklusiven oberen Rand => max+1.
inline uint16_t lbtRandMs(Core& ll, uint16_t minMs, uint16_t maxMs) {
  if (maxMs <= minMs) return minMs;
  const int32_t lo = (int32_t)minMs;
  const int32_t hiExclusive = (int32_t)maxMs + 1;
  int32_t r = ll.radio
                ? ll.radio->random(lo, hiExclusive)      // PhysicalLayer::random
                : (int32_t)::random((long)lo, (long)hiExclusive);
  return (uint16_t)r;
}

inline void startRxNow(Core& ll, const Callbacks& cb, RxKind kind, uint32_t now) {
  if (ll.current != Mode::Idle) return;      // Guard
  if (!ll.radio) return;

  ll.radio->startReceive();
  ll.current = Mode::Rx;
  ll.rxKind  = kind;
  ll.rxCountWinStart = ll.rxCount;

  if (kind == RxKind::Timed) {
    const uint16_t w = (ll.reqRxMs ? ll.reqRxMs : ll.defaultRxMs);
    ll.rxWindowEndMs = now + w;
    if (cb.onRxWindowOpen) cb.onRxWindowOpen(w, cb.ctx);
  } else { // Continuous
    ll.rxWindowEndMs = 0;
    if (cb.onRxWindowOpen) cb.onRxWindowOpen(0, cb.ctx);
  }
}

inline void setDefaultIdle(Core& ll) {
  ll.defaultRxKind = RxKind::None;
  ll.defaultRxMs   = 0;
}

inline void setDefaultRxContinuous(Core& ll) {
  ll.defaultRxKind = RxKind::Continuous;
  ll.defaultRxMs   = 0; // echt kontinuierlich
}

inline void setDefaultRxTimed(Core& ll, uint16_t windowMs) {
  ll.defaultRxKind = RxKind::Timed;
  ll.defaultRxMs   = windowMs;
}

inline void requestRxTimed(Core& ll, uint16_t windowMs) {
  ll.reqRxKind = RxKind::Timed; ll.reqRxMs = windowMs;
}
inline void requestRxContinuous(Core& ll) {
  ll.reqRxKind = RxKind::Continuous; ll.reqRxMs = 0;
}
inline void cancelRxRequest(Core& ll) {
  ll.reqRxKind = RxKind::None; ll.reqRxMs = 0;
}

// One-slot TX scheduling (no copy if len fits; returns false if slot busy or oversize).
inline bool scheduleSend(Core& ll, const uint8_t* buf, uint8_t len) {
  if (ll.hasPending || len == 0 || len > sizeof(ll.txBuf)) return false;
  memcpy(ll.txBuf, buf, len);
  ll.txLen = len;
  ll.earliestTxAtMs = millis();

  if (ll.lbtEnable) {
    uint16_t maxMs = lbtBackoffMaxMs(ll);
    ll.earliestTxAtMs += lbtRandMs(ll, 5, maxMs);
    ll.txArb = TxArbiter::CadNeeded;
  } else {
    ll.txArb = TxArbiter::None;
  }

  ll.hasPending = true;
  return true;
}

inline bool scheduleSendWithBackoff(Core& ll, const uint8_t* buf, uint8_t len,
                                    uint16_t jitterMinMs, uint16_t jitterMaxMs) {
  if (!scheduleSend(ll, buf, len)) return false;

  if (ll.lbtEnable) {
    uint16_t maxMs = lbtBackoffMaxMs(ll);
    ll.earliestTxAtMs = millis() + lbtRandMs(ll, 5, maxMs);
    ll.txArb = TxArbiter::CadNeeded;
  } else {
    const int32_t lo = (int32_t)jitterMinMs;
    const int32_t hiExclusive = (int32_t)jitterMaxMs + 1;
    int32_t r = ll.radio
                  ? ll.radio->random(lo, hiExclusive)
                  : (int32_t)::random((long)lo, (long)hiExclusive);
    ll.earliestTxAtMs = millis() + (uint16_t)r;
    ll.txArb = TxArbiter::None;
  }
  return true;
}

inline bool scheduleSendThenRxWindow(Core& ll, const uint8_t* buf, uint8_t len, uint16_t rxMs) {
  // TODO: so noch ok oder muss an LBT angepasst werden? Wird aktuell nicht genutzt.
  if (!scheduleSend(ll, buf, len)) return false;
  requestRxTimed(ll, rxMs);
  return true;
}

// Small wrapper for building + scheduling a typed payload (using LoraProto).
template<typename PayloadT>
inline bool buildAndSchedule(Core& ll, const uint8_t my3[3], const uint8_t dst3[3],
                             uint8_t fullType, const PayloadT& p) {
  uint8_t out[sizeof(LoraProto::Header7) + sizeof(PayloadT)];
  uint8_t n = LoraProto::build(out, my3, dst3, fullType, p);
  return scheduleSend(ll, out, n);
}

inline bool buildEmptyAndSchedule(Core& ll, const uint8_t my3[3], const uint8_t dst3[3],
                                  uint8_t fullType) {
  uint8_t out[sizeof(LoraProto::Header7)];
  uint8_t n = LoraProto::build_empty(out, my3, dst3, fullType);
  return scheduleSend(ll, out, n);
}

// -------------------- The service pump (call in loop()) --------------------
inline void service(Core& ll, const Callbacks& cb) {
  const uint32_t now = millis();

  // (A) IRQ: TX-Done oder RX-Paket
  if (ll.dio1Flag) {
    ll.dio1Flag = false;

    // LBT: CAD-Done?
    if (ll.txArb == TxArbiter::CadPending) {
      int16_t rc = ll.radio->getChannelScanResult();
      if (rc == RADIOLIB_CHANNEL_FREE) {
        // freigegeben → ab jetzt normaler TX-Pfad (B) mit txArb=None
        ll.txArb = TxArbiter::None;
        ll.earliestTxAtMs = millis();  // sofort versuchen
      } else {
        // busy → kurzer Backoff und erneut CAD wenn Zeit erreicht
        ll.txArb = TxArbiter::CadNeeded;
        ll.earliestTxAtMs = millis() + lbtRandMs(ll, 5, lbtBackoffMaxMs(ll));
      }
      // kein return: (B) darf in diesem service()-Durchlauf noch senden/planen
    }

    if (ll.current == Mode::Tx) {
      if (ll.radio) ll.radio->finishTransmit();
      ll.current = Mode::Idle;
      ++ll.txCount;                          // nur erfolgreiche TX zählen
      if (cb.onTxDone) cb.onTxDone(cb.ctx);

      // Nach TX: erst expliziten Request bedienen, sonst Default
      if (ll.reqRxKind != RxKind::None) {
        startRxNow(ll, cb, ll.reqRxKind, now);
        ll.reqRxKind = RxKind::None; ll.reqRxMs = 0;
      } else if (ll.defaultRxKind != RxKind::None) {
        ll.reqRxMs = ll.defaultRxMs;
        startRxNow(ll, cb, ll.defaultRxKind, now);
      }
    }
    else if (ll.current == Mode::Rx) {
      if (ll.radio) {
        size_t len = ll.radio->getPacketLength();
        if (len >= sizeof(LoraProto::Header7)) {
          if (len > sizeof(ll.txBuf)) len = sizeof(ll.txBuf);
          uint8_t pkt[64]; if (len > sizeof(pkt)) len = sizeof(pkt);
          if (ll.radio->readData(pkt, len) == RADIOLIB_ERR_NONE) {
            ll.lastRssi = (int16_t)ll.radio->getRSSI(true);
            ll.lastSnr  = (int8_t) ll.radio->getSNR();
            ++ll.rxCount;
            if (cb.onRxPacket) cb.onRxPacket(pkt, (uint8_t)len, ll.lastRssi, ll.lastSnr, cb.ctx);
          }
        }
        // Rx fortsetzen je nach Art
        if (ll.rxKind == RxKind::Continuous) {
          ll.radio->startReceive();
        } else { // Timed
          if ((int32_t)(now - ll.rxWindowEndMs) < 0) {
            ll.radio->startReceive();
          }
        }
      }
    }
  }

  // (B) TX pending starten
  if (ll.hasPending && (int32_t)(now - ll.earliestTxAtMs) >= 0) {
    if (ll.current == Mode::Idle ||
        (ll.current == Mode::Rx && ll.rxKind == RxKind::Continuous)) {

      if (ll.current == Mode::Rx && ll.rxKind == RxKind::Continuous) {
        ll.radio->standby();
        ll.current = Mode::Idle;
        ll.rxKind  = RxKind::None;
        ll.rxWindowEndMs = 0;
      }

      if (ll.lbtEnable) {
        // CAD-Gate: je nach Arbiter-Status handeln
        if (ll.txArb == TxArbiter::CadNeeded) {
          ChannelScanConfig_t cfg{};
          cfg.cad.symNum   = RADIOLIB_SX126X_CAD_ON_4_SYMB;   // robuste Defaults RADIOLIB_SX126X_CAD_ON_4_SYMB RADIOLIB_SX126X_CAD_ON_2_SYMB
          cfg.cad.detPeak  = 10;
          cfg.cad.detMin   = 5;
          cfg.cad.exitMode = RADIOLIB_SX126X_CAD_GOTO_STDBY; // CAD_ONLY
          cfg.cad.timeout  = 0; // alt: 8 ms
          //cfg.cad.timeout  = 5000; // 5 ms
          //cfg.cad.irqFlags = RADIOLIB_IRQ_CAD_DETECTED | RADIOLIB_IRQ_TIMEOUT;
          cfg.cad.irqFlags = (RadioLibIrqFlags_t)(RADIOLIB_IRQ_CAD_DONE | RADIOLIB_IRQ_CAD_DETECTED);
          cfg.cad.irqMask  = (RadioLibIrqFlags_t)(RADIOLIB_IRQ_CAD_DONE | RADIOLIB_IRQ_CAD_DETECTED);
          //ll.radio->startChannelScan(cfg);
          ll.radio->startChannelScan();
          ll.txArb = TxArbiter::CadPending;
          return; // auf IRQ warten (A wertet aus)
        } else if (ll.txArb == TxArbiter::CadPending) {
          return; // läuft noch – nichts tun
        }
        // else: TxArbiter::None → durchfallen zum normalen TX
      }

      // normaler TX-Start (einzige Stelle im Code)
      if (ll.radio->startTransmit(ll.txBuf, ll.txLen) == RADIOLIB_ERR_NONE) {
        ll.current = Mode::Tx;
        ll.hasPending = false;
        if (cb.onTxStart) cb.onTxStart(cb.ctx);
      }
    }
  }

  // (C) RX-Request nur aus Idle bedienen
  // C1: Expliziten Request bedienen (nur aus Idle)
  if (ll.current == Mode::Idle && ll.reqRxKind != RxKind::None && !ll.hasPending) {
    startRxNow(ll, cb, ll.reqRxKind, now);
    ll.reqRxKind = RxKind::None; ll.reqRxMs = 0;
  }

  // C2: Default durchsetzen (ohne extra Flag)
  //  - Falls wir in RxContinuous sind, der Default aber ≠ Continuous ist:
  //    Continuous sofort beenden und in Idle bleiben (neuer Default wird in der nächsten Iteration geöffnet).
  if (ll.current == Mode::Rx && ll.rxKind == RxKind::Continuous &&
      ll.defaultRxKind != RxKind::Continuous) {
    if (ll.radio) ll.radio->standby();
    ll.current = Mode::Idle;
    ll.rxKind  = RxKind::None;
    ll.rxWindowEndMs = 0;
  }

  //  - Falls wir Idle sind (und kein expliziter Request ansteht): Default öffnen.
  if (ll.current == Mode::Idle && ll.reqRxKind == RxKind::None &&
      ll.defaultRxKind != RxKind::None && !ll.hasPending) {
    ll.reqRxMs = ll.defaultRxMs;               // für Timed
    startRxNow(ll, cb, ll.defaultRxKind, now);
  }

  // (D) RX-Timed: Fensterende prüfen
  if (ll.current == Mode::Rx && ll.rxKind == RxKind::Timed && ll.rxWindowEndMs != 0) {
    if ((int32_t)(now - ll.rxWindowEndMs) >= 0) {
      if (ll.radio) ll.radio->standby();
      ll.current = Mode::Idle;
      const uint16_t delta = ll.rxCount - ll.rxCountWinStart;
      if (cb.onRxWindowClosed) cb.onRxWindowClosed(delta, cb.ctx);
      ll.rxKind = RxKind::None;
      ll.rxWindowEndMs = 0;

      // Default (falls gesetzt) direkt öffnen
      if (ll.defaultRxKind != RxKind::None) {
        ll.reqRxMs = ll.defaultRxMs;
        startRxNow(ll, cb, ll.defaultRxKind, now);
      }
    }
  }
}

// -------------------- Address utilities --------------------
inline bool readEfuseMac6(uint8_t mac6[6]) {
#if defined(ESP_PLATFORM) || defined(ESP32)
  if (esp_read_mac(mac6, ESP_MAC_WIFI_STA) == ESP_OK) return true;
#else
  String m = WiFi.macAddress();
  if (m.length() == 17) {
    for (int i=0;i<6;i++) mac6[i] = strtoul(m.substring(3*i, 3*i+2).c_str(), nullptr, 16);
    return true;
  }
#endif
  return false;
}

inline void last3FromMac6(uint8_t out[3], const uint8_t mac6[6]) {
  out[0] = mac6[3]; out[1] = mac6[4]; out[2] = mac6[5];
}

  snprintf(out, 18, "%02X:%02X:%02X:%02X:%02X:%02X", m[0], m[1], m[2], m[3], m[4], m[5]);
} */

inline void mac6ToStr(const uint8_t mac6[6], char out[18]) {
  static const char HEX_lookup[] = "0123456789ABCDEF";
  for (int i=0,j=0;i<6;i++) {
    uint8_t v = mac6[i];
    out[j++] = HEX_lookup[(v>>4)&0xF];
    out[j++] = HEX_lookup[v&0xF];
    if (i<5) out[j++] = ':';
  }
  out[17] = '\0';
}

// -------------------- Receiver/Broadcast helpers --------------------
inline bool same3(const uint8_t a[3], const uint8_t b[3]) {
  return a[0]==b[0] && a[1]==b[1] && a[2]==b[2];
}

inline bool isBroadcast3(const uint8_t last3[3]) {
  return last3[0]==0xFF && last3[1]==0xFF && last3[2]==0xFF;
}

inline bool receiverMatches(const uint8_t receiver3[3], const uint8_t myLast3[3]) {
  return isBroadcast3(receiver3) || same3(receiver3, myLast3);
}
